```rust
    parser.set_language(&tree_sitter_python::language())

                });

        );
    }

    group.finish();
}
```

```rust
// xtask/src/baseline.rs

fn parse_criterion_output(output: &str) -> Result<HashMap<String, BenchmarkResult>> {
    // Instead of parsing text output, use Criterion's JSON export
    // Criterion saves results to target/criterion/*/base/estimates.json

    let criterion_dir = PathBuf::from("target/criterion");
    let mut benchmarks = HashMap::new();

    for entry in std::fs::read_dir(&criterion_dir)? {
        let entry = entry?;
        let estimates_path = entry.path().join("base/estimates.json");

        if estimates_path.exists() {
            let json = std::fs::read_to_string(&estimates_path)?;
            let estimates: CriterionEstimates = serde_json::from_str(&json)?;

            benchmarks.insert(
                entry.file_name().to_string_lossy().to_string(),
                BenchmarkResult {
                    mean_us: estimates.mean.point_estimate / 1000.0, // ns → µs
                    stddev_us: estimates.std_dev.point_estimate / 1000.0,
                    samples: estimates.mean.confidence_interval.confidence_level as usize,
                    memory_bytes: None, // Criterion doesn't track memory
                },
            );
        }
    }

    Ok(benchmarks)
}

#[derive(Deserialize)]
struct CriterionEstimates {
    mean: Estimate,
    std_dev: Estimate,
}

#[derive(Deserialize)]
struct Estimate {
    point_estimate: f64,
    confidence_interval: ConfidenceInterval,
}

#[derive(Deserialize)]
struct ConfidenceInterval {
    confidence_level: f32,
}
```

**Success Criteria**:

### Phase 3: Arena Allocator (Week 3)

**Goal**: Implement and validate arena allocation for tree nodes

#### Task 3.1: Design Arena API

**File**: `runtime2/src/arena.rs`

**Design**:
```rust
use std::cell::Cell;
use std::marker::PhantomData;
use std::ptr::NonNull;

/// Bump allocator arena for tree nodes
///
/// Memory layout:
/// [Chunk 1: [Node|Node|Node...]] -> [Chunk 2: [Node|Node...]] -> ...
///
/// Lifetime: Arena must outlive all nodes allocated from it
pub struct TreeArena {
    chunks: Vec<Box<[TreeNode]>>,
    current_chunk: usize,
    current_offset: Cell<usize>,
    chunk_size: usize,
}

/// Handle to a node allocated in the arena
///
/// Safety: NodeHandle is valid only while the arena exists
#[derive(Copy, Clone)]
pub struct NodeHandle {
    chunk_idx: u32,
    offset: u32,
}

impl TreeArena {
    pub fn new() -> Self {
        Self::with_chunk_size(1024) // 1024 nodes per chunk
    }

    pub fn with_chunk_size(chunk_size: usize) -> Self {
        TreeArena {
            chunks: vec![Self::allocate_chunk(chunk_size)],
            current_chunk: 0,
            current_offset: Cell::new(0),
            chunk_size,
        }
    }

    /// Allocate a new node in the arena
    ///
    /// Returns a handle that can be dereferenced while arena is alive
    pub fn allocate_node(&self, kind: u16, range: Range) -> NodeHandle {
        let offset = self.current_offset.get();

        // Check if we need a new chunk
        if offset >= self.chunk_size {
            self.allocate_new_chunk();
            return self.allocate_node(kind, range); // Retry
        }

        // Write node data
        let chunk = &self.chunks[self.current_chunk];
        // SAFETY: offset is < chunk_size due to check above
        unsafe {
            let node_ptr = chunk.as_ptr().add(offset) as *mut TreeNode;
            node_ptr.write(TreeNode::new(kind, range));
        }

        // Advance offset
        self.current_offset.set(offset + 1);

        NodeHandle {
            chunk_idx: self.current_chunk as u32,
            offset: offset as u32,
        }
    }

    /// Get a reference to a node by handle
    ///
    /// Safety: Handle must be valid (allocated from this arena and not freed)
    pub fn get(&self, handle: NodeHandle) -> &TreeNode {
        &self.chunks[handle.chunk_idx as usize][handle.offset as usize]
    }

    /// Reset arena for reuse (incremental parsing)
    ///
    /// Safety: All existing NodeHandles become invalid after reset
    pub unsafe fn reset(&mut self) {
        self.current_chunk = 0;
        self.current_offset.set(0);
        // Chunks are reused, not deallocated
    }

    fn allocate_chunk(size: usize) -> Box<[TreeNode]> {
        // Allocate uninitialized chunk
        // Nodes are written on-demand in allocate_node
        let mut chunk = Vec::with_capacity(size);
        unsafe {
            chunk.set_len(size);
        }
        chunk.into_boxed_slice()
    }

    fn allocate_new_chunk(&mut self) {
        let new_chunk = Self::allocate_chunk(self.chunk_size);
        self.chunks.push(new_chunk);
        self.current_chunk += 1;
        self.current_offset.set(0);
    }
}

/// Tree node stored in arena
#[repr(C)]
pub struct TreeNode {
    kind: u16,
    range: Range,
    children: SmallVec<[NodeHandle; 4]>, // Inline storage for small node counts
}
```

**Success Criteria**:
- [ ] Arena compiles with no unsafe errors under `cargo miri test`

- [ ] `reset()` properly invalidates old handles (tested with ASAN)

**Probability**: Medium

**Mitigation**:

- Incremental optimization with frequent validation
- Focus on highest-impact hotspots first

**Fallback**:
- If ≤2× TS-C not achieved, document actual ratio
- Adjust targets for v0.9.0
- Investigate if GLR overhead is fundamental or optimizable

### Medium Risk: Arena Lifetime Complexity

**Probability**: Medium

**Mitigation**:
- Start with simple design (single arena per parser)
- Validate with Miri and ASAN continuously
- Consider Pin<> if self-referential issues arise

**Fallback**:
- Keep direct allocation as fallback feature flag
- Ship with arena as opt-in in v0.8.0, default in v0.9.0
